#include <WiFi.h>
#include <PubSubClient.h>

// WiFi and MQTT Configuration
const char* ssid = "Amr";
const char* password = "Amrmagdy33@";
const char* mqtt_server = "192.168.142.37"; // Raspberry Pi's IP
const char* mqtt_username = "Amr";
const char* mqtt_password = "Amrmagdy33@";
const char* mqtt_topic = "gaze/direction";

// Motor Pins
const int enA = 13;
const int in1 = 12;
const int in2 = 14;
const int enB = 25;
const int in3 = 27;
const int in4 = 26;

// Motor speed settings
const int motorSpeed = 75; // Base speed (0-255)
const int turnSpeed = 75;  // Turning speed

WiFiClient espClient;
PubSubClient client(espClient);
String currentGaze = "NO_FACE"; // Initialize with NO_FACE to match Raspberry Pi
unsigned long lastMsgTime = 0;

void setup_wifi() {
  delay(10);
  Serial.println();
  Serial.print("Connecting to ");
  Serial.println(ssid);

  WiFi.begin(ssid, password);

  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }

  Serial.println("");
  Serial.println("WiFi connected");
  Serial.println("IP address: ");
  Serial.println(WiFi.localIP());
}

void callback(char* topic, byte* payload, unsigned int length) {
  Serial.print("Message arrived [");
  Serial.print(topic);
  Serial.print("] ");
  
  currentGaze = "";
  for (int i = 0; i < length; i++) {
    currentGaze += (char)payload[i];
  }
  
  Serial.println(currentGaze);
  controlMotors(currentGaze);
}

void reconnect() {
  while (!client.connected()) {
    Serial.print("Attempting MQTT connection...");
    if (client.connect("ESP32Client", mqtt_username, mqtt_password)) {
      Serial.println("connected");
      client.subscribe(mqtt_topic);
    } else {
      Serial.print("failed, rc=");
      Serial.print(client.state());
      Serial.println(" try again in 5 seconds");
      delay(5000);
    }
  }
}

void setup() {
  // Initialize motor control pins
  pinMode(enA, OUTPUT);
  pinMode(enB, OUTPUT);
  pinMode(in1, OUTPUT);
  pinMode(in2, OUTPUT);
  pinMode(in3, OUTPUT);
  pinMode(in4, OUTPUT);
  
  // Initialize serial communication
  Serial.begin(115200);
  setup_wifi();
  client.setServer(mqtt_server, 1883);
  client.setCallback(callback);
  
  // Stop motors initially
  stopMotors();
}

void loop() {
  if (!client.connected()) {
    reconnect();
  }
  client.loop();
  
  // Safety stop if no messages received for 1 second
  if (millis() - lastMsgTime > 1000) {
    stopMotors();
    lastMsgTime = millis();
  }
}

void controlMotors(String direction) {
  lastMsgTime = millis();
  
  if (direction == "CENTER" || direction == "NO_FACE" || direction == "EYES_CLOSED") {
    stopMotors();
  } 
  else if (direction == "UP") {
    // Move forward
    digitalWrite(in1, HIGH);
    digitalWrite(in2, LOW);
    digitalWrite(in3, HIGH);
    digitalWrite(in4, LOW);
    analogWrite(enA, motorSpeed);
    analogWrite(enB, motorSpeed);
  } 
  else if (direction == "DOWN") {
    // Move backward
    digitalWrite(in1, LOW);
    digitalWrite(in2, HIGH);
    digitalWrite(in3, LOW);
    digitalWrite(in4, HIGH);
    analogWrite(enA, motorSpeed);
    analogWrite(enB, motorSpeed);
  } 
  else if (direction == "LEFT") {
    // Turn left
    digitalWrite(in1, LOW);
    digitalWrite(in2, HIGH);  // Left motor backward
    digitalWrite(in3, HIGH);  // Right motor forward
    digitalWrite(in4, LOW);
    analogWrite(enA, turnSpeed);
    analogWrite(enB, turnSpeed);
  } 
  else if (direction == "RIGHT") {
    // Turn right
    digitalWrite(in1, HIGH);  // Left motor forward
    digitalWrite(in2, LOW);
    digitalWrite(in3, LOW);   // Right motor backward
    digitalWrite(in4, HIGH);
    analogWrite(enA, turnSpeed);
    analogWrite(enB, turnSpeed);
  }
}

void stopMotors() {
  digitalWrite(in1, LOW);
  digitalWrite(in2, LOW);
  digitalWrite(in3, LOW);
  digitalWrite(in4, LOW);
  analogWrite(enA, 0);
  analogWrite(enB, 0);
}
