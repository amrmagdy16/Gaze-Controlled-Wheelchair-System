#include <WiFi.h>
#include <PubSubClient.h>

// WiFi and MQTT Configuration
const char* ssid = "Amr";
const char* password = "Amrmagdy33@";
const char* mqtt_server = "192.168.142.37";
const char* mqtt_username = "Amr";
const char* mqtt_password = "Amrmagdy33@";
const char* mqtt_topic = "gaze/direction";

// BTS7960 Motor Driver Pins
// Left Motor
const int L_RPWM = 12;  // Forward PWM
const int L_LPWM = 14;  // Reverse PWM
const int L_R_EN = 13;  // Forward Enable
const int L_L_EN = 27;  // Reverse Enable

// Right Motor
const int R_RPWM = 25;  // Forward PWM
const int R_LPWM = 26;  // Reverse PWM
const int R_R_EN = 32;  // Forward Enable
const int R_L_EN = 33;  // Reverse Enable

// Motor speed settings
const int motorSpeed = 130; // Base speed (0-255)
const int turnSpeed = 130;  // Turning speed

WiFiClient espClient;
PubSubClient client(espClient);
String currentGaze = "NO_DETECTION"; // Initialize with no detection
unsigned long lastMsgTime = 0;
const unsigned long detectionTimeout = 1000; // Stop after 1s of no messages

void setup_wifi() {
  delay(10);
  Serial.println();
  Serial.print("Connecting to ");
  Serial.println(ssid);

  WiFi.begin(ssid, password);

  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }

  Serial.println("");
  Serial.println("WiFi connected");
  Serial.println("IP address: ");
  Serial.println(WiFi.localIP());
}

void callback(char* topic, byte* payload, unsigned int length) {
  Serial.print("Message arrived [");
  Serial.print(topic);
  Serial.print("] ");
  
  currentGaze = "";
  for (int i = 0; i < length; i++) {
    currentGaze += (char)payload[i];
  }
  
  Serial.println(currentGaze);
  lastMsgTime = millis(); // Update last received message time
}

void reconnect() {
  while (!client.connected()) {
    Serial.print("Attempting MQTT connection...");
    if (client.connect("ESP32Client", mqtt_username, mqtt_password)) {
      Serial.println("connected");
      client.subscribe(mqtt_topic);
    } else {
      Serial.print("failed, rc=");
      Serial.print(client.state());
      Serial.println(" try again in 5 seconds");
      delay(5000);
    }
  }
}

void setup() {
  // Initialize motor control pins
  pinMode(L_RPWM, OUTPUT);
  pinMode(L_LPWM, OUTPUT);
  pinMode(L_R_EN, OUTPUT);
  pinMode(L_L_EN, OUTPUT);
  pinMode(R_RPWM, OUTPUT);
  pinMode(R_LPWM, OUTPUT);
  pinMode(R_R_EN, OUTPUT);
  pinMode(R_L_EN, OUTPUT);

  // Enable both directions (HIGH = active)
  digitalWrite(L_R_EN, HIGH);
  digitalWrite(L_L_EN, HIGH);
  digitalWrite(R_R_EN, HIGH);
  digitalWrite(R_L_EN, HIGH);

  // Initialize serial communication
  Serial.begin(115200);
  setup_wifi();
  client.setServer(mqtt_server, 1883);
  client.setCallback(callback);
  
  // Stop motors initially
  stopMotors();
}

void loop() {
  if (!client.connected()) {
    reconnect();
  }
  client.loop();
  
  // Check if detection is lost (no messages for timeout period)
  if (millis() - lastMsgTime > detectionTimeout) {
    currentGaze = "NO_DETECTION";
    stopMotors();
  }
  else {
    // Process gaze commands
    processGazeCommand();
  }
}

void processGazeCommand() {
  if (currentGaze == "CENTER") {
    // Move forward when centered
    analogWrite(L_RPWM, 0);
    analogWrite(L_LPWM, motorSpeed);
    analogWrite(R_RPWM, 0);
    analogWrite(R_LPWM, motorSpeed);
  } 
  else if (currentGaze == "DOWN") {
    // Move backward
    analogWrite(L_RPWM, 0);
    analogWrite(L_LPWM, motorSpeed);
    analogWrite(R_RPWM, 0);
    analogWrite(R_LPWM, motorSpeed);
  } 
  else if (currentGaze == "LEFT") {
    // Turn left
    analogWrite(L_RPWM, 0);
    analogWrite(L_LPWM, turnSpeed);
    analogWrite(R_RPWM, turnSpeed);
    analogWrite(R_LPWM, 0);
  } 
  else if (currentGaze == "RIGHT") {
    // Turn right
    analogWrite(L_RPWM, turnSpeed);
    analogWrite(L_LPWM, 0);
    analogWrite(R_RPWM, 0);
    analogWrite(R_LPWM, turnSpeed);
  }
  else if (currentGaze == "EYES_CLOSED" || currentGaze == "NO_DETECTION") {
    stopMotors();
  }
  // "UP" is intentionally ignored as per requirements
}

void stopMotors() {
  // Stop both motors
  analogWrite(L_RPWM, 0);
  analogWrite(L_LPWM, 0);
  analogWrite(R_RPWM, 0);
  analogWrite(R_LPWM, 0);
}
